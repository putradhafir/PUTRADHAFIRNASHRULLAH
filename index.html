3. Constructor


class Car {
  String? model;
  String? color;

  Car(this.model, this.color);

  void start() {
    print('Car started!');
  }
}

4. Validasi Data


class Mahasiswa {
  String nama;
  String nim;
  double ipk;

  Mahasiswa(this.nama, this.nim, this.ipk) {
    if (ipk < 0.0 || ipk > 4.0) {
      throw ArgumentError('IPK harus 0.0 - 4.0');
    }
  }
}

5. Encapsulation


class Person {
  String _name;
  int age;

  Person(this._name, this.age);

  void setName(String name) {
    _name = name;
  }

  String getName() {
    return _name;
  }
}

6. Named Parameter Constructor


class Produk {
  final String id;
  final String nama;
  final double harga;
  final int stok;
  final String kategori;

  Produk({
    required this.id,
    required this.nama,
    required this.harga,
    this.stok = 0,
    this.kategori = 'Umum',
  });
}
